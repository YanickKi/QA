\section{Simulated und quantum annealing}

\begin{frame}{Simulated annealing (SA)}
  \pause
  \begin{itemize}
    \setlength\itemsep{1em}
    \item in Analogie zu auskühlenden Metallen
    \begin{itemize}
      \pause
      \item schneller auskühlen \textrightarrow suboptimalerer Zustand (amorph)
      \pause
      \item langsamer auskühlen \textrightarrow optimalerer Zustand (kristallin)
    \end{itemize}
    \pause
    \item nutzt thermische Fluktuationen, um lokalen Minima zu entkommen
    \pause
    \item Wahrscheinlichkeit $P_i$ folgen Mastergleichung $\frac{dP_i}{dt} = \sum_j L_{ij} P_j$
    \pause
    \begin{itemize}
      \item        $L_{ij} = 
      \begin{cases}
        \left [ 1 + \text{exp}((E_i -E_j) / T_t) \right ]^{-1}  & \, \text{einzelne Spin-Differenz} \\
        - \sum_{k \neq i} L_{ki}                                & \, i = j \\
        0                                                       & \, \text{sonst}
      \end{cases}
    $
    \end{itemize}
  \end{itemize}

\begin{frame}{Quantum Annealing (QA)}
  \begin{itemize}
    \pause
    \item transversales Ising-Modell $H (t) = \underbrace{- \sum_{ij} J_{ij} \sigma_i^z \sigma_j^z - h \sum_i \sigma_i^z }_{H_\text{f}} \underbrace{- \Gamma (t) \sum_i \sigma_i^x}_{H_\text{i}(t)}  $ \pause
    \item $H_x (t)$ verursacht Tunneln zwischen Eigenzuständen von $H_0$ \pause
    \begin{textblock*}{5cm}(10.4cm, 2.05cm) % {block width} (coords)
      \includegraphics[width = 5cm]{Plots/tunneling.jpg}
      \hspace*{12pt}\hbox{\scriptsize {\footnotesize\itshape \href{https://doi.org/10.1016/j.physrep.2012.10.002}{V. Bapst et al., \textit{Physics Reports} 523 (2013)}}}
    \end{textblock*}
    \begin{itemize}
      \pause
      \item schnellere Konvergenz
    \end{itemize}
    \pause
    \item $\ket{\Psi(t)}$ durch Schrödingergleichung $i\frac{\partial \ket{\Psi(t)}}{\partial t} = H(t) \ket{\Psi(t)}$  \\ festgelegt
    \pause
    \item wähle monoton fallende Funktionen für $\Gamma(t)$ (\enquote{ausglühen})
  \end{itemize} 
  \vspace*{1cm}
  \pause
  \begin{block}{Adiabatisches Theorem der Quantenmechanik}
    \begin{center}
     System bleibt im selben Eigenzustand von $H$, wenn sich $H$ hinreichend langsam ändert  
    \end{center}    
  \end{block}

\end{frame}  

\begin{frame}{Adiabatische Zeit}
   \begin{itemize}
    \setlength\itemsep{1em}
    \pause
    \item minimale Zeit, damit sich System adiabatisch ändert $\tau$ \pause
    \item $\symup{\Delta} (s) = E_1 (s) - E_0 (s)$\pause
    \item mindestens $\tau \propto \frac{1}{\symup{\Delta} (s)^2}$, schlimmstenfalls $\tau \propto \frac{1}{\symup{\Delta} (s)^3}$\pause
    \item Problem: $\symup{\Delta} (s)$ sehr klein\pause
    \begin{itemize}
      \item[\textrightarrow] $\tau \to \infty$ 
    \end{itemize}
   \end{itemize}

  \end{equation*}

    \item $P_\text{SA}(t)$: Wahrscheinlichkeit System in GZ zu finden
    \item $P_\text{QA}(t)$ = $|\braket{g}{\Psi(t)}|^2$: Wahrscheinlichkeit System in GZ $\ket{g}$ von $H_f$ zu finden
    \item $P_\text{SA}^{\text{st}}(T)$: Boltzmann-Faktor des GZ von $H_f$
    \item $P_\text{QA}^{\text{st}}(\Gamma) = | \braket{g}{\Psi_{\Gamma}} |^2$ mit $\ket{\Psi_{\Gamma}}$ als GZ von $H$
    \item $\lim_{t \to 0} \Gamma(t) = \lim_{t \to 0} T(t) = \infty$
    \pause
    \begin{itemize}
      \item[\textrightarrow] Superposition aller Zustände gleicher Amplitude (QA)
      \item[\textrightarrow] Alle Zustände gleich wahrscheinlich (SA) 
    \end{itemize} \pause
    \item Idealfall (adiabatisch): $\lim_{t \to \infty} P_{QA}(t) = \lim_{t \to \infty} P_{SA}(t) = 1$ \pause
    \begin{itemize}
      \item[\textrightarrow] Grundzustand gefunden \pause
      \item[\textrightarrow] sonst: In einem lokalen Minimum \enquote{hängen} geblieben \pause
    \end{itemize}
    \item 8 Spins 
  \end{itemize}

\begin{frame}{Binäre Variable}
  \pause
  \begin{itemize}
    \setlength\itemsep{1em}
    \item $x \in \{0,1\}$ \pause
    \item Beispiel: Aufenthaltsort von einem Truck
    \item Formulierung als Graph mit $N$ Knoten $i$ \pause
    \begin{itemize}
      \item ordne jedem Knoten $i$ eine binäre Variable zu \pause
      \item jedem Truck kann ein Vektor $x^j = (x^j_1, \ldots, x^j_N)$ von binären Variablen zugeordnet werden 
    \end{itemize}
    \item Truck $j$ an Knoten $i$ \textrightarrow $x^j_i = 1$ sonst $x^j_k = 0$ 
  \end{itemize} 
\end{frame}

\begin{frame}{Quadratic unconstrained binary optimization (QUBO)} \pause
  \begin{block}{Aufgabe}
    \center
     Minimiere $ f(\underline{x}, \underline{\underline{Q}}) = \sum_{ij} Q_{ij} x_ix_j$ \pause
  \end{block}
  \begin{itemize}
    \setlength\itemsep{1em}
    \item $x_i \in \{0,1\}$ als binäre Variable \pause
    \begin{itemize}
      \item lässt sich mittels $s_i = 2x_i - 1 \in \{-1, 1\}$ auf Ising-Variablen (\enquote{Spin-1/2}) transformieren \pause
    \end{itemize}
    \item $Q_{ij}$ definiert Wechselwirkung zwischen $x_i$ \pause
    \begin{itemize}
      \item allg. hermitesch (symmetrisch im Ising-Modell)
    \end{itemize}
  \end{itemize}
\end{frame}
 
\begin{frame}{Quadratic unconstrained binary optimization constraints} \pause
  \begin{itemize}
    \item constraints lassen sich nicht direkt in der Hardware implementieren \pause
    \begin{itemize}
      \item constraints als \enquote{Strafe} in der Kostenfunktion implementieren \pause
    \end{itemize}
    \item Beispiel: Ein Truck darf sich nur an einem Punkt aufhalten \pause
    \begin{itemize} 
      \item $\sum_{i} x_i = 1 \iff \sum_i x_i - 1 = 0$ \pause
      \item in QUBO Problem umwandeln \textrightarrow $g(x_i) = (\sum_i x_i - 1)^2 $ \pause
      \item[\textrightarrow] $f \to f + \lambda g$ \pause
      \item constraint verletzt \textrightarrow Kostenfunktion erhöht \pause
    \end{itemize}
    \item Lagrangeparameter $\lambda$ sehr groß wählen
    \begin{itemize}
      \item[\textrightarrow] constraint eingehalten
    \end{itemize} 
  \end{itemize}

  \end{textblock*}
\end{frame}

    \setlength\itemsep{1em}
    \item D-Wave's QPU kann als Graph $U$ gesehen werden \pause
    \begin{itemize}
      \item Gewichtete Vertices (qubit) $i \in V(U)$ \pause
      \item Gewichtete Kanten (coupler) $ij \in E(U)$ \pause
    \end{itemize}
    \item Gewichte: \pause
    \begin{itemize}
      \item $h_i (t)$ \enquote{qubit bias}
      \item $\symup{\Delta}_i(t)$ \enquote{tunneling amplitude}
      \item $J_{ij}(t)$ \enquote{coupler strength}
    \end{itemize} \pause
    \item formuliere transversalen Ising-Hamiltonian auf Graph $G$ 
    \item $H(t) = \sum_{i \in V(G)} h_i(t) \sigma_i^z + \sum_{ij \in E(G)} J_{ij}(t) \sigma_i^z \sigma_j^z + \sum_{i \in V(G)} \Delta_i (t) \sigma_i^x$ \pause
    \item Eigenergie des Ising-Hamiltonians $\epsilon(s_1, \ldots, s_n) = \sum_{i \in V(G)} h_i s_i + \sum_{ij \in E(G)} J_{ij} s_i s_j$ \pause
    \begin{itemize}
      \item Spins $s_i \in \{-1,1\}$ \pause
    \end{itemize}
    \item Suche nach Grundzustandsenergie ist das \enquote{Ising-Problem} 
  \end{itemize}

\begin{frame}{Minor embedding}
  \begin{itemize}
    \item $\epsilon^\text{emb}(s_1, \ldots, s_N) = \sum_{i \in V(G_\text{emb})} h_i's_i + \sum_{ij \in E(G_\text{emb})} J_{ij}' s_i s_j $ minimieren \pause
    \item $\tau : V(G) \times V(G) \to V(U)$, so dass $i_{\tau (i,j)} \in V(T_i), \; j_{\tau (j,i)} \in V(T_j)$ mit $i_{\tau (i,j)} \, j_{\tau (j,i)} \in E(U)$ \pause
    \item $\text{OE}(G_\text{emb}) = \cup_{ij \in E(G)} i_{\tau (i,j)} \, j_{\tau (j,i)}$ sind ursprüngliche coupler \pause
    \begin{itemize}
      \item[\textrightarrow] $J_{i_{\tau (i,j)} \, j_{\tau (j,i)}}' = J_{ij}$ für $i_{\tau (i,j)} \, j_{\tau (j,i)} \in \text{OE}(G_\text{emb})$ \pause
    \end{itemize}
    \item ferromagnetische coupler strength $F_k^{ij}$ für $ij \in E(T_i)$ \pause
    \item bias $h_i$ der logischen qubits auf physische qubits verteilen  \pause
    \begin{itemize}
      \item[\textrightarrow] $\sum_{i_k \in V(T_i)} h_{i_k}' = h_i$ \pause
    \end{itemize}
  \end{itemize}
  \begin{block}{Finale Energie}
    \begin{equation*}
      \epsilon^{\text{emb}}(s_1, \ldots, s_N) = \sum_{i \in V(G)} \left (
        \sum_{i_k \in V(T_i)} h_{i_k}' s_{i_k} + \sum_{i_p i_q \in E(T_i)} F_i^{pq} s_{i_p} s_{i_q} \right ) 
        + \sum_{ij \in E(G)} J_{ij} s_{i_{\tau(i,j)}} s_{j_{\tau(i,j)}}
    \end{equation*}
  \end{block}

\section{Zusammenfassung und Ausblick}
\begin{frame}{Zusammenfassung und Ausblick}
  \begin{itemize}
    \item QA bereits angewandt um Logistik (deutlich) zu verbessern
    \item QA konvergiert wahrscheinlicher als SA
    \item nur auf kombinatorische Optimierungsprobleme anwendbar
    \item nur eine heuristische Methode
    \item Komplexität wächst bei QA exponentiell mit Systemgröße
  \end{itemize}
  \vspace*{1cm}
  \begin{itemize}
    \item mehr qubits 
    \begin{itemize}
      \item[\textrightarrow] größere Probleme optimieren
    \end{itemize}
    \item mehr coupler
    \begin{itemize}
      \item[\textrightarrow] weniger \enquote{dummy}-Vertices nötig
      \item[\textrightarrow] weniger physische qubits für einen logischen qubit nötig
    \end{itemize}
  \end{itemize}
\end{frame}
